import requests

def get_data(url):
    response = requests.get(url)
    return response.json()

jobs_data = get_data(url)
print("Data loaded. Number of job records:", len(jobs_data))
# Output: Data loaded. Number of job records: 27005
def count_jobs_for_skill(data, skill):
    count = 0
    for job in data:
        skills = job.get("Key Skills", "")
        if skill.lower() in skills.lower():
            count += 1
    return count
def count_jobs_by_location(data, location):
    count = 0
    for job in data:
        job_location = job.get("Location", "")
        if location.lower() in job_location.lower():
            count += 1
    return count
from openpyxl import Workbook

technologies = ["C", "C#", "C++", "Java", "JavaScript", "Python",
                "Scala", "Oracle", "SQL Server", "mysql", "PostgreSQL", "MongoDB"]

workbook = Workbook()
sheet = workbook.active
sheet.title = "Job Postings"
sheet.append(["Technology", "Job Count"])

for tech in technologies:
    job_count = count_jobs_for_skill(jobs_data, tech)
    sheet.append([tech, job_count])

workbook.save("job-postings.xlsx")

Key Code Snippets

fulltime_df = df[df['Employment'] == 'Employed, full-time']
print(f"Full-time median: ${fulltime_df['ConvertedCompYearly'].median():,.2f}")  # $69,814
print(f"Overall median:   ${df['ConvertedCompYearly'].median():,.2f}")           # $65,000
Q1, Q3 = df['ConvertedCompYearly'].quantile([0.25, 0.75])
IQR = Q3 - Q1
df_clean = df[(df['ConvertedCompYearly'] >= Q1 - 1.5*IQR) &
              (df['ConvertedCompYearly'] <= Q3 + 1.5*IQR)].copy()
print(f"Cleaned rows: {df_clean.shape[0]}")  # 22,457
corr_df     = df_clean[['ConvertedCompYearly','WorkExp','JobSatPoints_1']].dropna()
corr_matrix = corr_df.corr(method='pearson')
sns.heatmap(corr_matrix, annot=True, fmt='.4f', cmap='coolwarm', square=True)

fig, axes = plt.subplots(1, 2, figsize=(16, 6))
sns.regplot(data=corr_df, x='WorkExp',        y='ConvertedCompYearly', ax=axes[0])
sns.regplot(data=corr_df, x='JobSatPoints_1', y='ConvertedCompYearly', ax=axes[1])
